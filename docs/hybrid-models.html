# Use autoplot to display the time series and the forecasting results.
# autoplot(forecast(hmod), xlab = "Days", ylab = "Daily new deaths",
#     main = "Combined forecast from different models")
hmod.fc.report <- as.data.frame(hmod.fc) %>%
  mutate(DATE = as.Date("2020-12-17") + 1:14)
names(hmod.fc.report) <- c("Y.Death", "Lo.80", "Hi.80", 
                           "Lo.95", "Hi.95", "DATE")

ggplot(Florida.ts, aes(DATE, Y.Death)) + 
  geom_line(aes(color = 'Reported')) + 
  labs(x = "Days", y = "Daily new deaths") + 
  # Add prediction intervals
  geom_ribbon(mapping = aes(x = DATE, 
                            y = Y.Death, 
                            ymin = Lo.95, 
                            ymax = Hi.95,
                            fill = '95% Prediction intervals'), 
              data = hmod.fc.report, alpha = 0.4) +
  # Add line for predicted values
  geom_line(mapping = aes(x = DATE, 
                          y = Y.Death,
                          colour = 'Predicted'),
            data = hmod.fc.report,
            key_glyph = "timeseries") + 
  scale_colour_manual(values = c(Reported = "grey60", 
                                 Predicted = "black")) +
  scale_fill_manual("", values = "grey50") +        
  guides(color = guide_legend(title = "Series"),
         fill =  guide_legend(title = "")) +
  theme_bw() +
  theme(legend.position = "bottom")
